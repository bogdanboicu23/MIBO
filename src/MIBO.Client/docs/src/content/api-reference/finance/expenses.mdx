# Expenses API

Track and categorize user expenses.

**Base path:** `/api/expenses`

<Callout type="info" title="Authentication Required">
  All expense endpoints require a valid JWT Bearer token in the `Authorization` header.
</Callout>

<SectionDivider />

<SectionHeader title="List Expenses" description="Retrieve paginated expenses with optional category and date filters" />

<ApiEndpoint
    method="GET"
    path="/api/expenses"
    description="Retrieve paginated expenses with optional category and date filters."
    params={[
        { name: "userId", type: "string", required: false, description: "Filter by user ID" },
        { name: "category", type: "string", required: false, description: "Category filter" },
        { name: "startDate", type: "string", required: false, description: "Start date (ISO 8601)" },
        { name: "endDate", type: "string", required: false, description: "End date (ISO 8601)" },
        { name: "skip", type: "number", required: false, description: "Pagination offset (default: 0)" },
        { name: "limit", type: "number", required: false, description: "Page size (default: 30)" },
        { name: "sortBy", type: "string", required: false, description: "Sort field" },
        { name: "order", type: "string", required: false, description: "Sort direction" },
    ]}
    response={`{
    "expenses": [
        {
            "id": 1,
            "userId": 42,
            "transactionId": 101,
            "category": "groceries",
            "amount": 45.00,
            "currency": "USD",
            "description": "Weekly groceries",
            "date": "2024-01-15T10:30:00Z",
            "vendor": "Fresh Market",
            "isRecurring": false,
            "tags": ["food", "weekly"],
            "notes": ""
        }
    ],
    "total": 85,
    "skip": 0,
    "limit": 30
}`}
/>

<SectionDivider />

<SectionHeader title="Get Expense by ID" description="Retrieve a single expense by its unique identifier" />

<ApiEndpoint
    method="GET"
    path="/api/expenses/{id}"
    description="Retrieve a single expense by ID."
    params={[
        { name: "id", type: "number", required: true, description: "Expense ID" },
    ]}
/>

<SectionDivider />

<SectionHeader title="Get User Expenses" description="Get all expenses for a specific user with optional filters" />

<ApiEndpoint
    method="GET"
    path="/api/expenses/user/{userId}"
    description="Get all expenses for a specific user."
    params={[
        { name: "userId", type: "number", required: true, description: "User ID" },
        { name: "category", type: "string", required: false, description: "Category filter" },
        { name: "startDate", type: "string", required: false, description: "Start date" },
        { name: "endDate", type: "string", required: false, description: "End date" },
    ]}
/>

<SectionDivider />

<SectionHeader title="Get Categories" description="List all available expense categories" />

<ApiEndpoint
    method="GET"
    path="/api/expenses/categories"
    description="List all available expense categories."
    response={`[
    "groceries",
    "dining",
    "transport",
    "shopping",
    "entertainment",
    "utilities",
    "rent",
    "healthcare",
    "education",
    "subscriptions",
    "other"
]`}
/>

<Callout type="tip" title="Category Filtering">
  Use the categories returned by this endpoint as valid values for the `category` filter parameter on other expense endpoints.
</Callout>
