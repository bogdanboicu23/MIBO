# Transactions API

Query and search financial transactions.

**Base path:** `/api/transactions`

<Callout type="info" title="Authentication Required">
  All transaction endpoints require a valid JWT Bearer token in the `Authorization` header.
</Callout>

<SectionDivider />

<SectionHeader title="List Transactions" description="Retrieve a paginated list of transactions with comprehensive filtering" />

<ApiEndpoint
    method="GET"
    path="/api/transactions"
    description="Retrieve a paginated list of transactions with comprehensive filtering options."
    params={[
        { name: "userId", type: "string", required: false, description: "Filter by user ID" },
        { name: "accountId", type: "string", required: false, description: "Filter by account ID" },
        { name: "skip", type: "number", required: false, description: "Pagination offset (default: 0)" },
        { name: "limit", type: "number", required: false, description: "Page size (default: 30)" },
        { name: "type", type: "string", required: false, description: "Transaction type: debit, credit, transfer" },
        { name: "category", type: "string", required: false, description: "Category filter" },
        { name: "startDate", type: "string", required: false, description: "Start date (ISO 8601)" },
        { name: "endDate", type: "string", required: false, description: "End date (ISO 8601)" },
        { name: "sortBy", type: "string", required: false, description: "Sort field" },

<SectionHeader title="Get User Transactions" description="Get all transactions for a specific user with optional filters" />

<SectionDivider />

<SectionHeader title="Search Transactions" description="Full-text search across descriptions, categories, and merchants" />

<ApiEndpoint
    method="GET"
    path="/api/transactions/search"
    description="Full-text search across transaction descriptions, categories, and merchants."
    params={[
        { name: "q", type: "string", required: true, description: "Search query (returns 400 if empty)" },
        { name: "skip", type: "number", required: false, description: "Pagination offset" },
        { name: "limit", type: "number", required: false, description: "Page size" },
    ]}
/>

<Callout type="warning" title="Search Query Required">
  The `q` parameter is required and must not be empty. An empty search query will return a `400 Bad Request` response.
</Callout>
