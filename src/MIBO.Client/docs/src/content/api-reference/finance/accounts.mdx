# Accounts API

Manage user bank accounts in the Finance Data Service.

**Base path:** `/api/accounts`

<Callout type="info" title="Authentication Required">
  All account endpoints require a valid JWT Bearer token in the `Authorization` header.
</Callout>

<SectionDivider />

<SectionHeader title="List Accounts" description="Retrieve a paginated list of accounts with optional filtering and sorting" />

<ApiEndpoint
    method="GET"
    path="/api/accounts"
    description="Retrieve a paginated list of accounts with optional filtering and sorting."
    params={[
        { name: "userId", type: "string", required: false, description: "Filter by user ID" },
        { name: "skip", type: "number", required: false, description: "Number of records to skip (default: 0)" },
        { name: "limit", type: "number", required: false, description: "Maximum records to return (default: 30)" },
        { name: "select", type: "string", required: false, description: "Comma-separated fields to include" },
        { name: "sortBy", type: "string", required: false, description: "Field to sort by" },
        { name: "order", type: "string", required: false, description: "Sort order: asc or desc" },
    ]}

<SectionHeader title="Get Account by ID" description="Retrieve a single account by its unique identifier" />

<ApiEndpoint
    method="GET"
    path="/api/accounts/{id}"
    description="Retrieve a single account by its ID."
    params={[
        { name: "id", type: "number", required: true, description: "Account ID" },
    ]}
    response={`{
    "id": 1,
    "accountNumber": "ACC-001234",
    "accountName": "Main Checking",

    "balance": 12450.00,

}`}
/>

<SectionDivider />

<SectionHeader title="Get Accounts by User" description="Retrieve all accounts belonging to a specific user" />

<ApiEndpoint
    method="GET"
    path="/api/accounts/user/{userId}"
    description="Retrieve all accounts belonging to a specific user."
    params={[
        { name: "userId", type: "number", required: true, description: "User ID" },
        { name: "skip", type: "number", required: false, description: "Pagination offset (default: 0)" },
        { name: "limit", type: "number", required: false, description: "Page size (default: 30)" },
    ]}
/>
